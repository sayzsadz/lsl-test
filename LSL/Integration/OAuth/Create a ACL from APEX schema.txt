Create a ACL from APEX schema

begin

dbms_network_acl_admin.create_acl (

          acl => 'aluat.xml', -- or any other name

          description => 'HTTP Access',

          principal => 'APEX_040200', -- the user name trying to access the network resource

          is_grant => TRUE,

          privilege => 'connect',

          start_date => null,

          end_date => null

);

 

 

end;
